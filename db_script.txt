CREATE TABLE Sensores(

    Id INTEGER PRIMARY KEY,
    Local TEXT

);
--
CREATE TABLE Registros(

    Id INTEGER PRIMARY KEY,
    Valor FLOAT,
    Tipo INTEGER,
    TimeStamp Integer,
    Sensor INTEGER,
    FOREIGN KEY (Sensor)
        REFERENCES Sensores(Id)
            ON DELETE CASCADE
            ON UPDATE CASCADE

);
--
CREATE TABLE Transmissions(

    Id INTEGER PRIMARY KEY,
    TimeStamp INTEGER,
    Sensor INTEGER UNIQUE,
    FOREIGN KEY(Sensor)
        REFERENCES Sensores(Id)
            ON DELETE CASCADE
            ON UPDATE CASCADE

);
